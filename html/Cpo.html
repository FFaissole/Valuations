Section Mono.

Context {A : hSet}.
Context {LA : Le A}.
Context {B : hSet}.
Context {LB : Le B}.

Definition monotonic {OA : PartialOrder LA}
                     {OB : PartialOrder LB}
                     (f : A → B) := ∀ x y, x ≤ y → f x ≤ f y.

Record fmono {OA : PartialOrder LA}
             {OB : PartialOrder LB} : Type :=
              mk_fmono{ fmonot :> A → B;
                        fmonotonic: monotonic fmonot}.

Definition fmon {OA : PartialOrder LA}
                {OB : PartialOrder LB} : Le fmono.

End Mono.

Section Cpo.

Context {A : hSet}.
Context {LA : Le A}.
Context {B : hSet}.
Context {LB : Le B}.

Class cpo C {ordA : Le C} := mkcpo{
     cpobot : C;
     lub : ∀ (f : IncreasingSequence C), C;
     le_lub : ∀ (f : IncreasingSequence C) n,
                                      f n ≤ lub f;
     lub_le : ∀ (f : IncreasingSequence C) x,
               (∀ n, f n ≤ x) → lub f ≤ x ;
     cpobot_bot : ∀ x, cpobot ≤ x
}.

End Cpo.
