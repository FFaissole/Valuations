Definition of dominance
Definition dominance (Σ : Type0) (is_top : Σ → Type) :=
           ∀ (u:hProp) (s:Σ),
             (is_top s → hexists (fun z:Σ ⇒ is_top z = u)) →
             hexists (fun m:Σ ⇒ merely (is_top m)
                               = merely (u ∧ (is_top s))).
