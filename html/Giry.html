Giry monad on sets

unit operator
Definition unit (A : hSet) (x : A) : IntPos A.

bind operator
Definition bind (A B : hSet) : IntPos A → (A → IntPos B) → IntPos B.

Monadic properties
Lemma monad1 {A B : hSet} : ∀ (x : A) (F : A → IntPos B),
              bind (unit A x) F = F x.

Lemma monad2 {A : hSet} : ∀ (X : IntPos A),
                 bind X (unit A) = X.

Lemma monad3 {A B C: hSet} : ∀ (X : IntPos A)
                   (F : A → IntPos B) (G : B → IntPos C),
     bind (bind X F) G = bind X (fun x:A ⇒ bind (F x) G).

Lemma unit_mon {A :hSet} : ∀ (x:A) (f g : mf A),
    f ≤ g → (unit A x) f ≤ (unit A x) g.

Lemma bind_mon {A B :hSet} : ∀ (x:A) (X: IntPos A)
                       (F : A → IntPos B) (f g : mf B),
       f ≤ g → (bind X F) f ≤ (bind X F) g.
