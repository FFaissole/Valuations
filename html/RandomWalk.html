Definition Fiter : (nat → IntPos nat) → (nat → IntPos nat)
    := (fun f x ⇒ Iif flip (f (S x)) (unit nat x)).

Lemma Fiter_mon : ∀ a b, a ≤ b → Fiter a ≤ Fiter b.

Definition iterflip : nat → IntPos nat :=
                            Ifix Fiter Fiter_mon.
