Definition open_interval (p q : Q) : Cut → Type :=
                               fun x ⇒ QCut p < x ∧ x < QCut q.

Definition closed_interval (p q : Q) : Cut → Type :=
                               fun x ⇒ QCut p ≤ x ∧ x ≤ QCut q.
